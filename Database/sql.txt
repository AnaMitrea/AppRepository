CREATE TABLE Application (
	AppID INTEGER PRIMARY KEY AUTOINCREMENT,
	AppName TEXT,
	Developer TEXT,
	Executable_Name TEXT,
	License TEXT,
	Category TEXT,
	InternetConnection TEXT,
	AppInfo TEXT
);

CREATE TABLE OS (
	AppID INTEGER,
	OS_Name TEXT
);

CREATE TABLE Minimum_Req (
	AppID INTEGER,
	GHzCPU REAL,
	GPU TEXT,
	GB_RAM REAL,
	GB_HDStorage REAL
);
